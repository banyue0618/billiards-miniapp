#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

// --------------------- 配置区 ---------------------
const char* ssid = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

const char* mqtt_server = "192.168.1.100";  // EMQX 地址
const int mqtt_port = 1883;
const char* mqtt_client_id = "esp32-001";   // 每台设备唯一ID
const char* mqtt_user = "device001";        // EMQX 用户名
const char* mqtt_pass = "abc123";           // EMQX 密码

// 示例：billiards/{type}/{code}/command
const char* topic_sub = "billiards/light/001/command";       // 接收控制命令（建议 QoS1）

// 示例：billiards/{type}/{code}/status
const char* topic_pub_status = "billiards/light/001/status"; // 上线/离线/当前状态（retained）

WiFiClient espClient;
PubSubClient client(espClient);

const int RELAY_PIN = 5;  // 控制灯的继电器引脚
const bool RELAY_ACTIVE_HIGH = true; // 继电器触发电平，高/低电平有效
unsigned long lastStatusTime = 0;
const unsigned long STATUS_INTERVAL = 30 * 1000; // 每30秒上报状态

// --------------------- WiFi 连接 ---------------------
void connectWiFi() {
  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  int retry = 0;
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.print(".");
    if (++retry > 20) {   // 防止长时间卡住
      Serial.println("\nWiFi connect failed, restarting...");
      ESP.restart();
    }
  }
  Serial.println("\nWiFi connected!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
}

// --------------------- 消息回调 ---------------------
void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");

  // 提取字符串载荷（用于非JSON 0/1 指令）
  String raw;
  raw.reserve(length + 1);
  for (unsigned int i = 0; i < length; i++) raw += (char)payload[i];
  Serial.println(raw);

  // 继电器控制封装
  auto setRelay = [](bool on){
    digitalWrite(RELAY_PIN, (RELAY_ACTIVE_HIGH ? (on ? HIGH : LOW) : (on ? LOW : HIGH)));
  };

  // 解析：优先 JSON，其次纯文本（只接受 "1"/"0"）
  bool parsed = false;
  bool toOn = false;

  StaticJsonDocument<512> doc;
  DeserializationError error = deserializeJson(doc, raw);
  if (!error) {
    if (!parsed && doc.containsKey("command")) {
      if (doc["command"].is<int>()) { toOn = (doc["command"].as<int>() == 1); parsed = true; }
      else if (doc["command"].is<const char*>()) { String c = doc["command"].as<const char*>(); toOn = (c == "1" || c == "on" || c == "turn_on"); parsed = true; }
    }
  }

  if (!parsed) {
    String s = raw; s.trim(); s.toLowerCase();
    if (s == "1" || s == "on" || s == "turn_on") { toOn = true; parsed = true; }
    else if (s == "0" || s == "off" || s == "turn_off") { toOn = false; parsed = true; }
  }

  if (!parsed) {
    Serial.println("⚠️ 未识别命令");
    return;
  }

  setRelay(toOn);
  Serial.println(toOn ? "💡 灯已打开" : "💡 灯已关闭");

  // 执行完命令后上报当前状态（retained=true，便于新订阅者立即获知）
  StaticJsonDocument<256> ack;
  ack["status"] = (digitalRead(RELAY_PIN) == (RELAY_ACTIVE_HIGH ? HIGH : LOW)) ? 1 : 0;
  char buffer[256];
  size_t n = serializeJson(ack, buffer);
  client.publish(topic_pub_status, (uint8_t*)buffer, n, true);
}

// --------------------- MQTT 重连 ---------------------
void reconnectMQTT() {
  while (!client.connected()) {
    Serial.print("Connecting to MQTT...");
  // 设置遗嘱：离线状态（retained），online: 1/0 表示在线/离线
    bool ok = client.connect(
      mqtt_client_id, mqtt_user, mqtt_pass,
      topic_pub_status, 1 /*qos*/, true /*retained*/, "{\"online\":0}"
    );
    if (ok) {
      Serial.println("connected ✅");
      client.subscribe(topic_sub, 1); // QoS1 订阅
      client.publish(topic_pub_status, "{\"online\":1}", true); // retained 在线
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" retry in 5 seconds...");
      delay(5000);
    }
  }
}

// --------------------- 初始化 ---------------------
void setup() {
  Serial.begin(115200);
  pinMode(RELAY_PIN, OUTPUT);
  // 上电默认关闭
  digitalWrite(RELAY_PIN, (RELAY_ACTIVE_HIGH ? LOW : HIGH));

  connectWiFi();

  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);
  client.setKeepAlive(20);
  client.setBufferSize(1024);
  client.setSocketTimeout(4);
  WiFi.setSleep(false);
}

// --------------------- 主循环 ---------------------
void loop() {
  if (!client.connected()) {
    reconnectMQTT();
  }
  client.loop();

  // 定期上报心跳状态
  if (millis() - lastStatusTime > STATUS_INTERVAL) {
    StaticJsonDocument<256> doc;
    doc["deviceCode"] = mqtt_client_id; // 设备唯一ID
    doc["clientId"] = mqtt_client_id;
    doc["status"] = (digitalRead(RELAY_PIN) == HIGH) ? 1 : 0;
    doc["ip"] = WiFi.localIP().toString();
    doc["uptime"] = millis();
    char msg[256];
    size_t n = serializeJson(doc, msg);
    client.publish(topic_pub_status, (uint8_t*)msg, n, false); // 心跳非 retained
    lastStatusTime = millis();
  }
}
